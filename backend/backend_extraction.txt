import os
import re
import json
import traceback
import requests
import pandas as pd
import yfinance as yf
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from typing import List, Dict, Any, Optional
from backtesting import Backtest, Strategy
from datetime import datetime

router = APIRouter()

class Message(BaseModel):
    role: str
    content: str

class SimulatorRequest(BaseModel):
    symbol: str
    mode: str
    prompt: str
    chat_history: List[Message]
    parameters: Dict[str, Any]
    llm_provider: str
    api_key: str

def call_llm(provider: str, api_key: str, messages: List[Dict[str, str]]) -> str:
    if provider == "openai":
        url = "https://api.openai.com/v1/chat/completions"
        headers = {"Authorization": f"Bearer {api_key}", "Content-Type": "application/json"}
        payload = {"model": "gpt-4o", "messages": messages, "temperature": 0.3}
        res = requests.post(url, headers=headers, json=payload)
        res.raise_for_status()
        return res.json()["choices"][0]["message"]["content"]
        
    elif provider == "anthropic":
        url = "https://api.anthropic.com/v1/messages"
        headers = {"x-api-key": api_key, "anthropic-version": "2023-06-01", "Content-Type": "application/json"}
        system_msg = next((m["content"] for m in messages if m["role"] == "system"), "")
        user_msgs = [m for m in messages if m["role"] != "system"]
        payload = {"model": "claude-3-5-sonnet-20240620", "max_tokens": 2048, "system": system_msg, "messages": user_msgs, "temperature": 0.3}
        res = requests.post(url, headers=headers, json=payload)
        res.raise_for_status()
        return res.json()["content"][0]["text"]
        
    elif provider == "gemini":
        url = f"https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-pro:generateContent?key={api_key}"
        headers = {"Content-Type": "application/json"}
        gemini_msgs = [{"role": "user" if m["role"] in ["user", "system"] else "model", "parts": [{"text": m["content"]}]} for m in messages]
        payload = {"contents": gemini_msgs}
        res = requests.post(url, headers=headers, json=payload)
        res.raise_for_status()
        return res.json()["candidates"][0]["content"]["parts"][0]["text"]
        
    raise ValueError("Unsupported LLM Provider")

@router.post("/run")
async def run_simulation(req: SimulatorRequest):
    if not req.api_key:
        raise HTTPException(status_code=400, detail="API Key is required.")

    system_prompt = f"""
    You are an expert quantitative trading assistant. 
    Your goal is to help the user design and refine a trading strategy.
    
    CRITICAL RULE: DO NOT mention the underlying Python libraries, frameworks, or code (like `backtesting`, `pandas`, etc.) in your conversation.

    CURRENT BACKTEST PARAMETERS (Configured by user in the UI):
    - Target Symbol: {req.symbol}
    - Date Range: {req.parameters.get('startDate')} to {req.parameters.get('endDate')}
    - Initial Capital: ${req.parameters.get('initialCapital')}
    - Commission: {req.parameters.get('commission')} (as a fraction)

    PARAMETER EDITING:
    If the user explicitly asks to change parameters (e.g., "change initial capital to 50000" or "set date to 2024-01-01"), you MUST output a JSON block wrapped in ```json ... ``` with ONLY the updated keys. 
    Valid keys: "startDate" (YYYY-MM-DD), "endDate" (YYYY-MM-DD), "initialCapital" (number), "commission" (number).

    WORKFLOW:
    1. Converse with the user to understand their desired strategy rules.
    2. BEFORE writing any code, you MUST present a summary of the strategy AND the current parameters.
    3. Ask the user for explicit approval: "Should I run this backtest now?"
    4. ONLY when the user explicitly approves (e.g., "yes", "run it", "go ahead"), output the Python code wrapped in ```python ... ```.

    CODING RULES (when approved, invisible to user):
    - You MUST create a class named `CustomStrategy` that inherits from `Strategy`.
    - Use `self.data.Close`, `self.data.Open`, etc. 
    - Use `self.buy()`, `self.sell()`, `self.position.is_long`, etc.
    - DO NOT include data fetching or `Backtest()` execution calls. Only output the imports and the class.
    
    INDICATOR RULES (CRITICAL):
    - The environment ALREADY provides `SMA(values, n)`, `EMA(values, n)`, and `RSI(values, n)`. You can use them directly like `self.ema = self.I(EMA, self.data.Close, 30)`. DO NOT redefine them.
    - For ANY OTHER indicator (e.g., Bollinger Bands, MACD), you MUST define a helper function BEFORE the strategy class.
    - Inside your helper functions, `values` will be a NumPy array. You MUST wrap inputs in `pd.Series(values)` before applying pandas methods.
    """
    
    messages = [{"role": "system", "content": system_prompt}]
    for msg in req.chat_history:
        messages.append({"role": msg.role, "content": msg.content})
    messages.append({"role": "user", "content": req.prompt})

    try:
        llm_response = call_llm(req.llm_provider, req.api_key, messages)
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"LLM API Error: {str(e)}")

    # Parameter extraction
    param_update = None
    json_match = re.search(r'```json(.*?)```', llm_response, re.DOTALL)
    if json_match:
        try:
            param_update = json.loads(json_match.group(1).strip())
            llm_response = re.sub(r'```json.*?```', '', llm_response, flags=re.DOTALL).strip()
        except:
            pass

    code_match = re.search(r'```python(.*?)```', llm_response, re.DOTALL)
    if not code_match:
        return {"status": "chat_reply", "message": llm_response, "param_update": param_update}
    
    strategy_code = code_match.group(1).strip()

    try:
        start_date = req.parameters.get("startDate", "2023-01-01")
        end_date = req.parameters.get("endDate", datetime.today().strftime('%Y-%m-%d'))
        df = yf.download(req.symbol, start=start_date, end=end_date, progress=False)
        if df.empty:
            raise ValueError(f"No data fetched for {req.symbol}")
        if isinstance(df.columns, pd.MultiIndex):
            df.columns = df.columns.get_level_values(0)
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"Data fetch error: {str(e)}")

    exec_env = {}
    setup_code = """
import pandas as pd
import numpy as np
from backtesting import Strategy
from backtesting.lib import crossover

def SMA(values, n):
    return pd.Series(values).rolling(n).mean()

def EMA(values, n):
    return pd.Series(values).ewm(span=n, adjust=False).mean()

def RSI(values, n):
    delta = pd.Series(values).diff()
    gain = (delta.where(delta > 0, 0)).rolling(window=n).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(window=n).mean()
    rs = gain / loss
    return 100 - (100 / (1 + rs))
"""
    try:
        exec(setup_code, exec_env, exec_env)
        exec(strategy_code, exec_env, exec_env)
        CustomStrategy = exec_env.get("CustomStrategy")
        if not CustomStrategy:
            raise ValueError("LLM did not produce a class named 'CustomStrategy'")
    except Exception as e:
        error_msg = f"Failed to compile strategy code.\nError: {str(e)}\n\nCode generated:\n{strategy_code}"
        return {"status": "error", "message": error_msg, "param_update": param_update}

    try:
        cash = float(req.parameters.get("initialCapital", 100000))
        comm = float(req.parameters.get("commission", 0.0))
        
        bt = Backtest(df, CustomStrategy, cash=cash, commission=comm, exclusive_orders=True)
        stats = bt.run()
        
        trades_df = stats['_trades']
        trade_markers = []
        for _, row in trades_df.iterrows():
            size = int(row['Size'])
            direction = "Buy" if size > 0 else "Sell"
            trade_markers.append({
                "time": int(row['EntryTime'].timestamp()),
                "type": direction,
                "price": row['EntryPrice'],
                "size": abs(size)
            })
            trade_markers.append({
                "time": int(row['ExitTime'].timestamp()),
                "type": "Exit",
                "price": row['ExitPrice'],
                "pnl": row['PnL'],
                "size": abs(size)
            })
            
        equity_curve = stats['_equity_curve']
        equity_data = []
        for dt, row in equity_curve.iterrows():
            equity_data.append({"time": int(dt.timestamp()), "value": row['Equity']})

        return {
            "status": "success",
            "message": "Strategy executed successfully.",
            "code": strategy_code,
            "param_update": param_update,
            "results": {
                "return_pct": stats['Return [%]'],
                "win_rate": stats['Win Rate [%]'],
                "max_drawdown": stats['Max. Drawdown [%]'],
                "sharpe": stats['Sharpe Ratio'],
                "trades": trade_markers,
                "equity_curve": equity_data
            }
        }

    except Exception as e:
        return {"status": "error", "message": f"Backtest execution error: {traceback.format_exc()}", "param_update": param_update}
