@dataclass
class Cmd:
    cmd_id: str
    type: str
    payload: Dict[str, Any] = None 
    job_id: Optional[str] = None

    def __post_init__(self):
        if self.payload is None:
            self.payload = {}

def safe_float(val: Any) -> Optional[float]:
    try:
        if val is None or val == "": return None
        f = float(val)
        return f if math.isfinite(f) else None
    except (ValueError, TypeError):
        return None

def is_valid_bar(bar: Dict[str, Any]) -> bool:
    required = ["open", "high", "low", "close"]
    return all(bar.get(k) is not None for k in required)

def to_epoch_seconds(x) -> int:
    if isinstance(x, (int, float)): return int(x)
    if isinstance(x, datetime): return int(x.replace(tzinfo=timezone.utc).timestamp())
    if isinstance(x, date): return int(datetime.combine(x, dtime.min).replace(tzinfo=timezone.utc).timestamp())
    return 0

def reply(r: redis.Redis, cmd_id: str, status: str, payload: Any = None, error: str = None):
    msg = json.dumps({"cmd_id": cmd_id, "status": status, "payload": payload, "error": error})
    r.xadd(f"{REPLY_PREFIX}{cmd_id}", {"json": msg}, maxlen=1000)

def publish_ib_status(r: redis.Redis, connected: bool, error: str = None):
    data = json.dumps({"connected": connected, "last_heartbeat": time.time(), "error": error})
    r.set(STATUS_KEY, data, ex=30)

def handle_get_account_summary(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    account_id = cmd.payload.get("account_id")
    try:
        if account_id:
            try: ib.client.reqAccountUpdates(True, account_id)
            except Exception as e: pass
            
        ib.sleep(0.05) 
        
        account_values = ib.accountValues(account_id) if account_id else ib.accountValues()
        values = {av.tag: (safe_float(str(av.value)) if str(av.value).replace('.','').replace('-','').isdigit() else str(av.value)) for av in account_values}
        summary = {
            "account_id": account_id,
            "net_liquidation": values.get("NetLiquidation", 0),
            "excess_liquidity": values.get("ExcessLiquidity", 0),
            "maintenance_margin": values.get("MaintMarginReq", 0),
            "available_funds": values.get("AvailableFunds", 0),
            "buying_power": values.get("BuyingPower", 0),
            "total_cash": values.get("TotalCashValue", 0),
            "unrealized_pnl": values.get("UnrealizedPnL", 0),
            "realized_pnl": values.get("RealizedPnL", 0),
            "currency": values.get("Currency", "USD")
        }
        reply(r, cmd.cmd_id, "OK", payload={"summary": summary})
    except Exception as e:
        reply(r, cmd.cmd_id, "ERROR", error=str(e))

def handle_get_positions(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    account_id = cmd.payload.get("account_id")
    try:
        ib.client.reqPositions()
        if account_id:
            try: ib.client.reqAccountUpdates(True, account_id)
            except Exception as e: pass
            
        ib.sleep(0.05) 
        
        portfolio = ib.portfolio(account_id) if account_id else []
        raw_positions = [p for p in ib.positions() if not account_id or p.account == account_id]
        
        positions = []
        portfolio_map = {item.contract.conId: item for item in portfolio}
        
        if raw_positions:
            contracts_to_req = [p.contract for p in raw_positions]
            ticker_map = {}
            if contracts_to_req:
                ib.qualifyContracts(*contracts_to_req)
                tickers = ib.reqTickers(*contracts_to_req)
                ticker_map = {t.contract.conId: t for t in tickers}

            for p in raw_positions:
                port_item = portfolio_map.get(p.contract.conId)
                ticker = ticker_map.get(p.contract.conId)
                
                mkt_price = 0.0
                daily_pnl = None
                
                if ticker:
                    val = ticker.marketPrice()
                    mkt_price = val if not math.isnan(val) else 0.0
                    
                    if not math.isnan(ticker.close) and ticker.close > 0 and mkt_price > 0:
                        daily_pnl = (mkt_price - ticker.close) * p.position

                if port_item:
                    cost_basis = port_item.averageCost * port_item.position
                    unrealized_pnl = port_item.unrealizedPNL
                    unrealized_pnl_pct = (unrealized_pnl / abs(cost_basis)) * 100 if cost_basis != 0 else 0
                    
                    port_mkt_val = safe_float(port_item.marketValue)
                    calc_mkt_val = (mkt_price * p.position)
                    final_mkt_val = port_mkt_val if port_mkt_val is not None and port_mkt_val != 0 else calc_mkt_val

                    positions.append({
                        "account_id": p.account,
                        "symbol": p.contract.symbol,
                        "sec_type": p.contract.secType,
                        "exchange": p.contract.exchange or p.contract.primaryExchange,
                        "currency": p.contract.currency,
                        "position": p.position,
                        "avg_cost": port_item.averageCost,
                        "market_price": mkt_price if mkt_price > 0 else port_item.marketPrice,
                        "market_value": final_mkt_val,
                        "unrealized_pnl": safe_float(unrealized_pnl) or 0.0,
                        "unrealized_pnl_pct": round(unrealized_pnl_pct, 2),
                        "realized_pnl": safe_float(port_item.realizedPNL) or 0.0,
                        "daily_pnl": daily_pnl 
                    })
                else:
                    cost_basis = p.avgCost * p.position
                    mkt_value = mkt_price * p.position
                    unrealized_pnl = mkt_value - cost_basis if mkt_price > 0 else 0.0
                    unrealized_pnl_pct = (unrealized_pnl / abs(cost_basis)) * 100 if cost_basis != 0 and mkt_price > 0 else 0.0

                    positions.append({
                        "account_id": p.account,
                        "symbol": p.contract.symbol,
                        "sec_type": p.contract.secType,
                        "exchange": p.contract.exchange or p.contract.primaryExchange,
                        "currency": p.contract.currency,
                        "position": p.position,
                        "avg_cost": p.avgCost,
                        "market_price": mkt_price,
                        "market_value": mkt_value, 
                        "unrealized_pnl": unrealized_pnl,
                        "unrealized_pnl_pct": round(unrealized_pnl_pct, 2),
                        "realized_pnl": 0.0,
                        "daily_pnl": daily_pnl 
                    })
        else:
            contracts_to_req = [item.contract for item in portfolio]
            ticker_map = {}
            if contracts_to_req:
                ib.qualifyContracts(*contracts_to_req)
                tickers = ib.reqTickers(*contracts_to_req)
                ticker_map = {t.contract.conId: t for t in tickers}

            for item in portfolio:
                ticker = ticker_map.get(item.contract.conId)
                daily_pnl = None
                
                port_mkt_price = safe_float(item.marketPrice)
                mkt_price = port_mkt_price or 0.0
                
                if ticker:
                    val = ticker.marketPrice()
                    mkt_price = val if not math.isnan(val) else mkt_price
                    if not math.isnan(ticker.close) and ticker.close > 0 and mkt_price > 0:
                        daily_pnl = (mkt_price - ticker.close) * item.position

                cost_basis = item.averageCost * item.position
                unrealized_pnl_pct = (item.unrealizedPNL / abs(cost_basis)) * 100 if cost_basis != 0 else 0
                
                port_mkt_val = safe_float(item.marketValue)
                final_mkt_val = port_mkt_val if port_mkt_val is not None and port_mkt_val != 0 else (mkt_price * item.position)

                positions.append({
                    "account_id": item.account,
                    "symbol": item.contract.symbol,
                    "sec_type": item.contract.secType,
                    "exchange": item.contract.exchange or item.contract.primaryExchange,
                    "currency": item.contract.currency,
                    "position": item.position,
                    "avg_cost": item.averageCost,
                    "market_price": mkt_price,
                    "market_value": final_mkt_val,
                    "unrealized_pnl": safe_float(item.unrealizedPNL) or 0.0,
                    "unrealized_pnl_pct": round(unrealized_pnl_pct, 2),
                    "realized_pnl": safe_float(item.realizedPNL) or 0.0,
                    "daily_pnl": daily_pnl 
                })
                
        reply(r, cmd.cmd_id, "OK", payload={"positions": positions})
    except Exception as e:
        print(f"[{IB_HOST}] Daemon Error (positions): {traceback.format_exc()}")
        reply(r, cmd.cmd_id, "ERROR", error=str(e))

def handle_get_orders(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    account_id = cmd.payload.get("account_id")
    status_filter = cmd.payload.get("status_filter")
    try:
        ib.client.reqAllOpenOrders() 
        ib.sleep(0.05) 
        
        all_trades = ib.trades() 
        orders_out = []
        for trade in all_trades:
            order, contract = trade.order, trade.contract
            if account_id and order.account != account_id: continue
            order_status = trade.orderStatus.status
            if status_filter:
                if status_filter == "open" and order_status in ["Filled", "Cancelled"]: continue
                if status_filter == "filled" and order_status != "Filled": continue
                if status_filter == "cancelled" and order_status != "Cancelled": continue
            orders_out.append({
                "order_id": order.orderId,
                "perm_id": order.permId,
                "account_id": order.account,
                "symbol": contract.symbol,
                "sec_type": contract.secType,
                "action": order.action,
                "order_type": order.orderType,
                "quantity": order.totalQuantity,
                "filled_quantity": trade.orderStatus.filled,
                "remaining_quantity": trade.orderStatus.remaining,
                "limit_price": order.lmtPrice if order.lmtPrice != 0 else None,
                "stop_price": order.auxPrice if order.auxPrice != 0 else None,
                "avg_fill_price": trade.orderStatus.avgFillPrice if trade.orderStatus.avgFillPrice != 0 else None,
                "status": order_status,
                "created_at": None
            })
        reply(r, cmd.cmd_id, "OK", payload={"orders": orders_out})
    except Exception as e:
        reply(r, cmd.cmd_id, "ERROR", error=str(e))

def handle_get_trades(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    account_id = cmd.payload.get("account_id")
    try:
        ib.client.reqExecutions(10001, ExecutionFilter())
        ib.sleep(0.05) 
        
        fills = ib.fills()
        trades = []
        for fill in fills:
            execution, contract, commission = fill.execution, fill.contract, fill.commissionReport
            if account_id and execution.acctNumber != account_id: continue
            trades.append({
                "exec_id": execution.execId,
                "order_id": execution.orderId,
                "account_id": execution.acctNumber,
                "symbol": contract.symbol,
                "sec_type": contract.secType,
                "action": execution.side,
                "quantity": execution.shares,
                "price": execution.price,
                "commission": commission.commission if commission else 0,
                "realized_pnl": commission.realizedPNL if commission else None,
                "executed_at": execution.time.isoformat() if execution.time else None,
                "exchange": execution.exchange
             })
        reply(r, cmd.cmd_id, "OK", payload={"trades": trades})
    except Exception as e:
        reply(r, cmd.cmd_id, "ERROR", error=str(e))

def handle_symbol_search(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    query = cmd.payload.get("query", "").upper()
    max_results = cmd.payload.get("max_results", 10)
    if not query: return reply(r, cmd.cmd_id, "OK", payload={"matches": []})
    try:
        descriptions = ib.reqMatchingSymbols(query)
        matches = []
        for d in descriptions[:max_results]:
            c = d.contract
            matches.append({
                "symbol": c.symbol,
                "name": c.description or c.localSymbol or c.symbol,
                "sec_type": c.secType,
                "exchange": c.primaryExchange or c.exchange or "SMART",
                "currency": c.currency,
                "description": c.description or c.localSymbol or c.symbol
            })
        reply(r, cmd.cmd_id, "OK", payload={"matches": matches})
    except Exception as e:
        reply(r, cmd.cmd_id, "ERROR", error=str(e))

# --- HELPER TO EXTRACT VALUES SAFELY ---
def get_first_valid(df, date_col, keys):
    if df is None or df.empty: return None
    for k in keys:
        if k in df.index:
            val = safe_float(df.loc[k, date_col])
            if val is not None:
                return val
    return None

def handle_get_profile(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    symbol = cmd.payload.get("symbol", "").upper()
    if not symbol: return reply(r, cmd.cmd_id, "ERROR", error="No symbol provided")
    
    try:
        contract = Stock(symbol, "SMART", "USD")
        ib.qualifyContracts(contract)
        cds = ib.reqContractDetails(contract)
        cd = cds[0] if cds else None
        
        profile_data = {
            "symbol": symbol,
            "companyName": cd.longName if cd else symbol,
            "industry": cd.industry if cd else None,
            "sector": cd.category if cd else None,
            "exchange": contract.primaryExchange if contract.primaryExchange else None,
            "employees": None,
            "summary": None,
            "pe": None,
            "eps": None,
            "dividendYield": None,
            "high52": None,
            "low52": None,
            "marketCap": None,
            "revenue": None,
            "targetPrice": None,
            "recommendation": None,
            "volume": None,
            "avgVolume": None,
            "nextEarningsDate": None, 
            "nextEarningsDays": None, 
            "beta": None,         
            "floatShares": None,   
            "earnings": [],
            "dividends": None,     
            "financials": {
                "income": {"annual": [], "quarterly": []},
                "balance": {"annual": [], "quarterly": []},
                "cashflow": {"annual": [], "quarterly": []}
            },
            "performance": {"1W": None, "1M": None, "3M": None, "6M": None, "YTD": None, "1Y": None},
            "seasonality": [],
            "analyst_ratings": {
                "consensus_text": None,
                "consensus_score": None,
                "target_price": None,
                "total_analysts": 0,
                "breakdown": {
                    "strong_buy": 0,
                    "buy": 0,
                    "hold": 0,
                    "sell": 0,
                    "strong_sell": 0
                }
            },
            "technicals": {"rsi": None, "sma": None, "signal": "NEUTRAL"},
            "news": []
        }

        try:
            ticker_obj = yf.Ticker(symbol)
            info = ticker_obj.info
            
            if info:
                profile_data["industry"] = info.get("industry", profile_data["industry"])
                profile_data["sector"] = info.get("sector", profile_data["sector"])
                profile_data["employees"] = info.get("fullTimeEmployees", profile_data["employees"])
                profile_data["summary"] = info.get("longBusinessSummary", profile_data["summary"])
                
                profile_data["revenue"] = info.get("totalRevenue")
                
                profile_data["targetPrice"] = info.get("targetMeanPrice")
                profile_data["analyst_ratings"]["target_price"] = info.get("targetMeanPrice")
                
                if info.get("recommendationKey"):
                    rec_key = str(info.get("recommendationKey")).replace("_", " ").title()
                    profile_data["recommendation"] = rec_key
                    profile_data["analyst_ratings"]["consensus_text"] = rec_key
                    
                profile_data["marketCap"] = info.get("marketCap")
                profile_data["pe"] = info.get("trailingPE", info.get("forwardPE"))
                
                div = info.get("dividendYield")
                if div: profile_data["dividendYield"] = div * 100
                    
                profile_data["high52"] = info.get("fiftyTwoWeekHigh")
                profile_data["low52"] = info.get("fiftyTwoWeekLow")
                profile_data["eps"] = info.get("trailingEps")
                
                if info.get("shortName"):
                    profile_data["companyName"] = info.get("shortName")

                profile_data["volume"] = info.get("volume", info.get("regularMarketVolume"))
                profile_data["avgVolume"] = info.get("averageVolume")
                profile_data["beta"] = info.get("beta")
                profile_data["floatShares"] = info.get("floatShares", info.get("sharesOutstanding"))

                # --- ACCURATE FUTURE EARNINGS DATE EXTRACTION ---
                earn_date = None
                cal = None
                try:
                    cal = getattr(ticker_obj, 'calendar', None)
                    if isinstance(cal, dict) and "Earnings Date" in cal and cal["Earnings Date"]:
                        earn_date = cal["Earnings Date"][0] 
                except Exception: pass
                
                if not earn_date:
                    earnings_ts = info.get("earningsTimestamp") or info.get("earningsTimestampStart")
                    if earnings_ts:
                        try:
                            earn_date = datetime.fromtimestamp(earnings_ts, tz=timezone.utc).date()
                        except Exception: pass

                if earn_date:
                    try:
                        today = datetime.now(timezone.utc).date()
                        diff = (earn_date - today).days
                        if diff >= 0:
                            profile_data["nextEarningsDate"] = earn_date.strftime("%B %d, %Y").replace(" 0", " ")
                            profile_data["nextEarningsDays"] = diff
                    except Exception: pass

                # --- DIVIDEND DATA EXTRACTION ---
                div_data = {
                    "payoutRatio": info.get("payoutRatio"),
                    "yieldTTM": info.get("trailingAnnualDividendYield", info.get("dividendYield")),
                    "lastPayment": info.get("lastDividendValue", info.get("dividendRate")),
                    "exDividendDate": None,
                    "paymentDate": None
                }
                
                ex_div_ts = info.get("exDividendDate")
                if ex_div_ts:
                    try:
                        div_data["exDividendDate"] = datetime.fromtimestamp(ex_div_ts, tz=timezone.utc).strftime("%b %d, %Y").replace(" 0", " ")
                    except Exception: pass
                    
                try:
                    if isinstance(cal, dict):
                        if not div_data["exDividendDate"] and "Ex-Dividend Date" in cal and cal["Ex-Dividend Date"]:
                            ex_date = cal["Ex-Dividend Date"]
                            if isinstance(ex_date, list) and len(ex_date) > 0: ex_date = ex_date[0]
                            div_data["exDividendDate"] = ex_date.strftime("%b %d, %Y").replace(" 0", " ")
                        if "Dividend Date" in cal and cal["Dividend Date"]:
                            pay_date = cal["Dividend Date"]
                            if isinstance(pay_date, list) and len(pay_date) > 0: pay_date = pay_date[0]
                            div_data["paymentDate"] = pay_date.strftime("%b %d, %Y").replace(" 0", " ")
                except Exception: pass
                
                profile_data["dividends"] = div_data
                    
            yf_news = ticker_obj.news
            if yf_news:
                for item in yf_news[:5]:
                    title = item.get('title')
                    link = item.get('link', '#')
                    pub_time = item.get('providerPublishTime')
                    
                    date_str = ""
                    if pub_time:
                        date_str = datetime.fromtimestamp(pub_time, tz=timezone.utc).isoformat()
                    
                    if title:
                        profile_data["news"].append({
                            "title": title,
                            "link": link,
                            "date": date_str
                        })

            # --- ANALYST RATINGS BREAKDOWN EXTRACTION ---
            try:
                recs = ticker_obj.recommendations
                if recs is not None and not recs.empty:
                    current_recs = recs[recs['period'] == '0m'] if 'period' in recs.columns else recs.head(1)
                    if not current_recs.empty:
                        sb = int(current_recs['strongBuy'].iloc[0])
                        b = int(current_recs['buy'].iloc[0])
                        h = int(current_recs['hold'].iloc[0])
                        s = int(current_recs['sell'].iloc[0])
                        ss = int(current_recs['strongSell'].iloc[0])
                        total = sb + b + h + s + ss
                        
                        profile_data["analyst_ratings"]["breakdown"] = {
                            "strong_buy": sb,
                            "buy": b,
                            "hold": h,
                            "sell": s,
                            "strong_sell": ss
                        }
                        profile_data["analyst_ratings"]["total_analysts"] = total
                        if total > 0:
                            score = ((sb * 5) + (b * 4) + (h * 3) + (s * 2) + (ss * 1)) / total
                            profile_data["analyst_ratings"]["consensus_score"] = round(score, 2)
            except Exception as e:
                print(f"Analyst breakdown extraction error: {e}")

            # --- TRADINGVIEW EXACT CALENDAR MATH (PERFORMANCE) ---
            try:
                hist = ticker_obj.history(period="2y")
                if not hist.empty:
                    current_close = safe_float(hist['Close'].iloc[-1])
                    current_date = hist.index[-1]
                    
                    if current_close:
                        def get_return_by_date(target_date):
                            past_data = hist[hist.index <= target_date]
                            if not past_data.empty:
                                past_close = safe_float(past_data['Close'].iloc[-1])
                                if past_close and past_close > 0:
                                    return ((current_close - past_close) / past_close) * 100
                            return None

                        profile_data["performance"]["1W"] = get_return_by_date(current_date - pd.Timedelta(days=7))
                        profile_data["performance"]["1M"] = get_return_by_date(current_date - pd.DateOffset(months=1))
                        profile_data["performance"]["3M"] = get_return_by_date(current_date - pd.DateOffset(months=3))
                        profile_data["performance"]["6M"] = get_return_by_date(current_date - pd.DateOffset(months=6))
                        profile_data["performance"]["1Y"] = get_return_by_date(current_date - pd.DateOffset(years=1))
                        
                        prev_year = current_date.year - 1
                        prev_year_end = pd.Timestamp(year=prev_year, month=12, day=31, tz=current_date.tz)
                        profile_data["performance"]["YTD"] = get_return_by_date(prev_year_end)
            except Exception as e:
                print(f"Performance extraction error: {e}")

            # --- SEASONALITY (DAILY CUMULATIVE RETURNS FOR CHART) ---
            try:
                hist_seas = ticker_obj.history(period="5y")
                if not hist_seas.empty:
                    years = sorted(hist_seas.index.year.unique(), reverse=True)[:3]
                    seasonality_data = []
                    
                    for year in years:
                        year_df = hist_seas[hist_seas.index.year == year]
                        if not year_df.empty:
                            first_close = safe_float(year_df['Close'].iloc[0])
                            if first_close and first_close > 0:

def handle_hist_bars(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    p = cmd.payload
    symbol = p["symbol"]
    try:
        contract = Stock(symbol, "SMART", "USD")

def handle_subscribe_bars(ib: IB, r: redis.Redis, cmd: Cmd) -> None:
    symbol = cmd.payload.get("symbol", "").upper()
    if symbol in LIVE_SUBSCRIPTIONS: 
        cb = LIVE_SUBSCRIPTIONS[symbol]["bar"]
        if cb.get("open") is not None:
            r.publish(f"{BARS_PREFIX}{symbol}", json.dumps(cb))
        return reply(r, cmd.cmd_id, "OK", payload={"status": "already_subscribed"})
        
    try:
        contract = Stock(symbol, "SMART", "USD")
        ib.qualifyContracts(contract)
        
        hist = ib.reqHistoricalData(
            contract, 
            endDateTime='', 
            durationStr='4 D', 
            barSizeSetting='1 day', 
            whatToShow='TRADES', 
            useRTH=1, 
            formatDate=1
        )
        
        robust_prev = None
        robust_last = None
        
        if hist:
            if len(hist) >= 2:
                robust_prev = safe_float(hist[-2].close) 
                robust_last = safe_float(hist[-1].close) 
            elif len(hist) == 1:
                robust_last = safe_float(hist[-1].close)
                robust_prev = safe_float(hist[-1].open)

        hist_ext = ib.reqHistoricalData(
            contract, 
            endDateTime='', 
            durationStr='1 D', 
            barSizeSetting='1 min', 
            whatToShow='TRADES', 
            useRTH=0, 
            formatDate=1
        )
        
        immediate_live_price = robust_last
        if hist_ext and len(hist_ext) > 0:
            ext_val = safe_float(hist_ext[-1].close)
            if ext_val is not None and ext_val > 0:
                immediate_live_price = ext_val
                
        ticker = ib.reqMktData(contract, '', False, False)
        
        now_ts = int(time.time())
        LIVE_SUBSCRIPTIONS[symbol] = {
            "ticker": ticker, 
            "bar": {
                "time": now_ts, 
                "open": immediate_live_price, 
                "high": immediate_live_price, 
                "low": immediate_live_price, 
                "close": immediate_live_price, 
                "volume": 0, 
                "prev_close": robust_prev,
                "rth_close": robust_last 
            }, 
            "last_pub": 0,
            "last_broadcast": 0 
        }
        
        r.publish(f"{BARS_PREFIX}{symbol}", json.dumps(LIVE_SUBSCRIPTIONS[symbol]["bar"]))
        
        def on_tick_update(t: Ticker):
            now = int(time.time())
            val = t.last if t.last is not None and not math.isnan(t.last) and t.last > 0 else None
            
            if val is None: return 
            
            price = safe_float(val)
            if price is None: return 
            
            sub = LIVE_SUBSCRIPTIONS.get(symbol)
            if not sub: return
            cb = sub["bar"]
                
            if now > cb["time"]:
                sub["bar"] = {
                    "time": now, 
                    "open": price, 
                    "high": price, 
                    "low": price, 
                    "close": price, 
                    "volume": 0, 
                    "prev_close": cb.get("prev_close"), 
                    "rth_close": cb.get("rth_close")    
                }
            else:
                cb["high"] = max(cb["high"] or price, price)
                cb["low"] = min(cb["low"] or price, price)
                cb["close"] = price
                
        ticker.updateEvent += on_tick_update
        reply(r, cmd.cmd_id, "OK", payload={"status": "subscribed"})
    except Exception as e:
        reply(r, cmd.cmd_id, "ERROR", error=f"IB Error: {e}")

def main():
    r = redis.Redis.from_url(REDIS_URL, decode_responses=True)
    ib = IB()
    print(f"[{IB_HOST}] Daemon started in {MODE.upper()} mode. Target: {IB_PORT}")

    last_hb = 0
    last_published_state = None
    last_reconnect_time = 0
    was_fully_connected = False
    
    try:
        last_id = r.xrevrange(CMD_STREAM, count=1)[0][0] if r.exists(CMD_STREAM) else "0-0"
    except:
        last_id = "0-0"
        
    while True:
        is_connected = ib.isConnected()
        
        if is_connected:
            was_fully_connected = True
        elif was_fully_connected:
            was_fully_connected = False
            if LIVE_SUBSCRIPTIONS: LIVE_SUBSCRIPTIONS.clear()

        if not is_connected:
            try:
                try: ib.disconnect()
                except: pass
                
                ib = IB()
                cid = random.randint(100, 9999)
                publish_ib_status(r, connected=False, error="offline")
                
                ib.connect(IB_HOST, IB_PORT, clientId=cid, timeout=3)
                
                is_connected = True
                was_fully_connected = True
                print(f"[{IB_HOST}] Connected!")
                
                try:
                    ib.reqMarketDataType(4)
                    ib.client.reqPositions()
                    ib.client.reqAllOpenOrders()
                    ib.client.reqExecutions(10001, ExecutionFilter())
                except Exception: pass
                
                publish_ib_status(r, connected=True, error=None)
                last_published_state = True
                last_hb = time.time()
            except Exception:
                time.sleep(1)
                is_connected = False
        
        if is_connected != last_published_state or (time.time() - last_hb > 1.5):
            publish_ib_status(r, connected=is_connected, error=None)
            last_published_state = is_connected
            last_hb = time.time()

        try:
            if is_connected: ib.sleep(0.02)
            else: time.sleep(0.02)
        except Exception:
            is_connected = False

        try:
            resp = r.xread({CMD_STREAM: last_id}, count=10, block=10)
            if resp:
                for _, entries in resp:
                    for entry_id, fields in entries:
                        last_id = entry_id
                        try:
                            cmd_json = fields.get("json", "{}")
                            cmd = Cmd(**json.loads(cmd_json))
                            if cmd.type == "RECONNECT": 
                                if time.time() - last_reconnect_time > 15:
                                    handle_reconnect(ib, r, cmd)
                                    last_reconnect_time = time.time()
                            elif cmd.type == "PING": reply(r, cmd.cmd_id, "OK", payload={"pong": time.time()})
                            elif not is_connected: reply(r, cmd.cmd_id, "ERROR", error="Gateway offline")
                            else:
                                if cmd.type == "HIST_BARS": handle_hist_bars(ib, r, cmd)
                                elif cmd.type == "SUBSCRIBE_BARS": handle_subscribe_bars(ib, r, cmd)
                                elif cmd.type == "SYMBOL_SEARCH": handle_symbol_search(ib, r, cmd)
                                elif cmd.type == "GET_PROFILE": handle_get_profile(ib, r, cmd)
                                elif cmd.type == "SERVER_TIME": handle_server_time(ib, r, cmd)
                                elif cmd.type == "GET_ACCOUNTS": handle_get_accounts(ib, r, cmd)
                                elif cmd.type == "GET_ACCOUNT_SUMMARY": handle_get_account_summary(ib, r, cmd)
                                elif cmd.type == "GET_POSITIONS": handle_get_positions(ib, r, cmd)
                                elif cmd.type == "GET_ORDERS": handle_get_orders(ib, r, cmd)
                                elif cmd.type == "GET_TRADES": handle_get_trades(ib, r, cmd)
                        except Exception as e:
                            if 'cmd' in locals() and cmd: reply(r, cmd.cmd_id, "ERROR", error=str(e))
        except Exception: pass
        
        if is_connected:
            now_int = int(time.time())
            for symbol, sub in list(LIVE_SUBSCRIPTIONS.items()):
                cb = sub["bar"]
                if cb.get("open") is not None:
                    tick_changed = (sub["last_pub"] != cb["time"])
                    heartbeat_due = (now_int - sub.get("last_broadcast", 0) >= 1)

                    if tick_changed or heartbeat_due:
                        r.publish(f"{BARS_PREFIX}{symbol}", json.dumps(cb))
                        sub["last_pub"] = cb["time"]
                        sub["last_broadcast"] = now_int

if __name__ == "__main__":
    main()

from __future__ import annotations
from dataclasses import dataclass, field
from typing import Dict, List
from .types import Fill, Prices


@dataclass
class Portfolio:
    cash: float
    positions: Dict[str, float] = field(default_factory=dict)
    fills: List[Fill] = field(default_factory=list)
    equity_curve: List[float] = field(default_factory=list)
    commission_paid: float = 0.0
    slippage_cost: float = 0.0


    def apply_fill(self, fill: Fill) -> None:
        self.positions[fill.symbol] = self.positions.get(fill.symbol, 0) + fill.qty
        self.cash -= fill.qty * fill.price
        self.fills.append(fill)

    def equity(self, prices: Prices) -> float:
        eq = self.cash
        for sym, qty in self.positions.items():
            if qty == 0:
                continue
            if sym not in prices:
                raise ValueError(f"Missing price for {sym}")
            eq += qty * prices[sym]
        return eq
    
    def apply_commission(self, amount: float) -> None:
        self.cash -= amount
        self.commission_paid += amount

    def apply_slippage_cost(self, amount: float) -> None:
        self.slippage_cost += amount
